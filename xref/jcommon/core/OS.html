134   public static OS fromName(final String Name) {
135     if (StringUtil.isNullOrEmpty(Name))
136       return OS.Unknown;
137 
138     for(OS os : OS.values()) {
139       for(String variant : os.variants)
140         if (variant.equalsIgnoreCase(Name))
141           return os;
142     }
143 
144     final String lower = Name.toLowerCase();
145     if (lower.contains("win"))
146       return OS.WindowsUnknown;
147     else if (lower.contains("mac"))
148       return OS.MacUnknown;
149     else if (lower.contains("nix") || lower.contains("nux"))
150       return OS.UnixUnknown;
151     else if (lower.contains("vms"))
152       return OS.VMSUnknown;
153     else if (lower.contains("solaris"))
154       return OS.SolarisUnknown;
155     else
156       return OS.Unknown;
157   }
