67    public static boolean isPOSIX(final OSFamily OSFamily) {
68      switch(OSFamily) {
69        case Unix:
70        case Mac:
71        case Solaris:
72          return true;
73        default:
74          return false;
75      }
76    }
77  
78    public static OSFamily fromName(final String Name) {
79      if (StringUtil.isNullOrEmpty(Name))
80        return OSFamily.Unknown;
81  
82      for(OSFamily family : OSFamily.values()) {
83        if (family.platformPartName.equalsIgnoreCase(Name))
84          return family;
85      }
86  
87      final String lower = Name.toLowerCase();
88      if (lower.contains("win"))
89        return OSFamily.Windows;
90      else if (lower.contains("mac"))
91        return OSFamily.Mac;
92      else if (lower.contains("nix") || lower.contains("nux"))
93        return OSFamily.Unix;
94      else if (lower.contains("vms"))
95        return OSFamily.VMS;
96      else if (lower.contains("solaris"))
97        return OSFamily.Solaris;
98      else
99        return OSFamily.Unknown;
100   }
