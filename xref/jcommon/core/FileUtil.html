20  package jcommon.core;
21  
22  import java.io.File;
23  import java.util.Stack;
24  
25  /**
26   * Contains general-purpose file and directory manipulation utilities.
27   */
28  public class FileUtil {
29    /**
30     * Uses an iterative algorithm to recursively delete an entire directory and all its subdirectories.
31     * Could be revised later for Java v1.7 compatibility by using Files.walkFileTree().
32     *
33     * @param path The directory to be deleted.
34     * @return <code>true</code> if the entire directory and all subdirectories were removed; <code>false</code> otherwise.
35     * @see <a href="http://fahdshariff.blogspot.com/2011/08/java-7-deleting-directory-by-walking.html">http://fahdshariff.blogspot.com/2011/08/java-7-deleting-directory-by-walking.html</a>
36     */
37    public static final boolean deleteDirectory(final File path) {
38      if (!path.isDirectory())
39        throw new IllegalArgumentException("path must be a directory");
40      if (path == null)
41        return false;
42      if (!path.exists())
43        return true;
44  
45      try {
46        File curr;
47        Stack<File> stack = new Stack<File>();
48        Stack<File> dir_stack = new Stack<File>();
49        stack.push(path);
50  
51        while(!stack.empty() && (curr = stack.pop()) != null) {
52          for(File potential : curr.listFiles()) {
53            if (potential.isDirectory()) {
54              stack.push(potential);
55              dir_stack.push(potential);
56            } else {
57              potential.delete();
58            }
59          }
60        }
61  
62        while(!dir_stack.empty() && (curr = dir_stack.pop()) != null) {
63          curr.delete();
64        }
65  
66        path.delete();
67  
68        return true;
69      } catch(Throwable t) {
70        return false;
71      }
72    }
73  }
